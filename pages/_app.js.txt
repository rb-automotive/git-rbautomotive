import type { AppProps } from 'next/app';

const roboto_slab = Roboto_Slab({ subsets: ['latin'], weight: '700', variable: '--font-roboto-slab' }); // Define CSS variable


function MyApp({ Component, pageProps }: AppProps) {
  return (
    // Apply font variables to the main div
    // Using font-sans which should map to Inter via tailwind.config.js
    <div className={`${inter.variable} ${roboto_slab.variable} flex flex-col min-h-screen font-sans`}>
      <Header />
      <main className="flex-grow">
        <Component {...pageProps} /> {/* Render the current page */}
      </main>
      <Footer />
    </div>
  );
}

export default MyApp;
